public class Main
{
	public static void main(String[] args) {
        Dwarf dwarf = new Dwarf("Test", 100);
        dwarf.sing();
        dwarf.sleep();
        System.out.println(dwarf.getHealthPoints());
        
        Animal animal = new Animal("Test", 1, 1);
        animal.feed();
        System.out.println(animal.getSize());
        
        Weapon weapon = new Weapon("Test", 10, 1);
        weapon.sharpen();
        System.out.println(weapon.getSize());
        System.out.println(weapon.getAttackSpeed());
	}
}

class Dwarf {
    String name;
    int healthPoints;

    public Dwarf(String name, int healthPoints) {
        this.name = name;
        this.healthPoints = healthPoints;
    }

    public String getName() {
        return this.name;
    }

    public int getHealthPoints() {
        return this.healthPoints;
    }

    public void setHealthPoints(int newHealthPoints) {
        if (newHealthPoints >= 0) {
            this.healthPoints = newHealthPoints;
        }
    }
    
    public void sing() {
        System.out.println("La-La-La");
    }
    
    public void sleep() {
        this.healthPoints = this.healthPoints + 1;
    }

}

class Weapon {
    int size;
    int attackSpeed;
    String name;

    public Weapon(String name, int size, int attackSpeed) {
        this.name = name;
        this.size = size;
        this.attackSpeed = attackSpeed;
    }

    public int getSize() {
        return this.size;
    }

    public String getName() {
        return this.name;
    }

    public int getAttackSpeed() {
        return this.attackSpeed;
    }

    public void setSize(int size) {
        this.size = size;
    }
    
    public void sharpen() {
        this.size = this.size - 1;
        this.attackSpeed = this.attackSpeed + 1;
    }

}

class Animal {
    int size;
    String name;
    int age;

    public Animal(String name, int size, int age) {
        this.name = name;
        this.size = size;
        this.age = age;
    }

    public int getSize() {
        return this.size;
    }

    public String getName() {
        return this.name;
    }

    public int getAge() {
        return this.age;
    }

    public void setAge(int age) {
        this.age = age;
    }
    
    public void feed() {
        this.size = this.size + 1;
    }

}
