import java.util.ArrayList;
import java.util.List;
import java.util.Random;

public class Main
{
	public static void main(String[] args) {
        Dwarf dwarf = new Dwarf("test", 100, null);

        System.out.println("Сила атаки дворфа: " + dwarf.getAttackPower());

        Weapon weapon = new Weapon("test", 10, 10);
        dwarf.setWeapon(weapon);
        System.out.println("Сила атаки дворфа после добавления оружия: " + dwarf.getAttackPower());

        List<Animal> animals = new ArrayList<>();
        Random random = new Random();

        for (int i = 0; i < 500; i++) {
            if (random.nextInt(2) == 0) {
                animals.add(new Dog("DOG" + i, random.nextInt(150) + 1, random.nextInt(150) + 1));
            } else {
                animals.add(new Cat("CAT" + i, random.nextInt(120)+ 1, random.nextInt(150) + 1));
            }
        }

        for (Animal animal : animals) {
            System.out.println(animal.sound());
	    }

        Sounder sounder = new Sounder();
        sounder.sound(new Cat("test", 1, 1));
        sounder.sound(new Dog("test", 1, 1));
    }
}

class Sounder {
    public void sound(Cat cat) {
        System.out.println("Кот говорит: " + cat.sound());
    }
    
    public void sound(Dog dog) {
        System.out.println("Собака говорит: " + dog.sound());
    }
}


class Dwarf {
    protected String name;
    protected int healthPoints;
    protected Weapon weapon;

    public Dwarf(String name, int healthPoints, Weapon weapon) {
        this.name = name;
        this.healthPoints = healthPoints;
        this.weapon = weapon;
    }

    public String getName() {
        return this.name;
    }

    public int getHealthPoints() {
        return this.healthPoints;
    }

    public void setHealthPoints(int newHealthPoints) {
        if (newHealthPoints >= 0) {
            this.healthPoints = newHealthPoints;
        }
    }
    
    public void sing() {
        System.out.println("La-La-La");
    }
    
    public void sleep() {
        this.healthPoints = this.healthPoints + 1;
    }

    public int getAttackPower() {
        int weaponSize = 1;
        if (this.weapon != null) {
            weaponSize = this.weapon.getSize();
        }
        return Math.round(this.healthPoints / 100 * weaponSize);
    }

    public void setWeapon(Weapon weapon) {
        this.weapon = weapon;
    }

}

class Weapon {
    protected int size;
    protected int attackSpeed;
    protected String name;

    public Weapon(String name, int size, int attackSpeed) {
        this.name = name;
        this.size = size;
        this.attackSpeed = attackSpeed;
    }

    public int getSize() {
        return this.size;
    }

    public String getName() {
        return this.name;
    }

    public int getAttackSpeed() {
        return this.attackSpeed;
    }

    public void setSize(int size) {
        this.size = size;
    }
    
    public void sharpen() {
        this.size = this.size - 1;
        this.attackSpeed = this.attackSpeed + 1;
    }

}

class Animal {
    protected int size;
    protected String name;
    protected int age;

    public Animal(String name, int size, int age) {
        this.name = name;
        this.size = size;
        this.age = age;
    }

    public int getSize() {
        return this.size;
    }

    public String getName() {
        return this.name;
    }

    public int getAge() {
        return this.age;
    }

    public void setAge(int age) {
        this.age = age;
    }
    
    public void feed() {
        this.size = this.size + 1;
    }

    public String sound() {
        return "";
    }       

}

class Dog extends Animal {

    public Dog(String name, int size, int age) {
        super(name, size, age);    
        }

    public void feed() {
        this.size = this.size + 2;
    }

    @Override
    public String sound() {
        return "Woof";
    }    

}

class Cat extends Animal {

    public Cat(String name, int size, int age) {
        super(name, size, age);    
    }

    @Override
    public String sound() {
        return "Murr";
    }    

}
