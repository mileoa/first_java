# 1.2. 3 реализация 3 классов
```java
class Dwarf {
    String name;
    int healthPoints;

    public Dwarf(String name, int healthPoints) {
        this.name = name;
        this.healthPoints = healthPoints;
    }

    public String getName() {
        return this.name;
    }

    public int getHealthPoints() {
        return this.healthPoints;
    }

    public void setHealthPoints(int newHealthPoints) {
        if (newHealthPoints >= 0) {
            this.healthPoints = newHealthPoints;
        }
    }

}

class Weapon {
    int size;
    int attackSpeed;
    String name;

    public Weapon(String name, int size, int attackSpeed) {
        this.name = name;
        this.size = size;
        this.attackSpeed = attackSpeed;
    }

    public int getSize() {
        return this.size;
    }

    public String getName() {
        return this.name;
    }

    public int getAttackSpeed() {
        return this.attackSpeed;
    }

    public void setSize(int size) {
        this.size = size;
    }

}

class Animal {
    int size;
    String name;
    int age;

    public Animal(String name, int size, int age) {
        this.name = name;
        this.size = size;
        this.age = age;
    }

    public int getSize() {
        return this.size;
    }

    public String getName() {
        return this.name;
    }

    public int getAge() {
        return this.age;
    }

    public void setAge(int age) {
        this.age = age;
    }

}

public class Main
{
	public static void main(String[] args) {
        Dwarf dwarf = new Dwarf("Test", 100);
        System.out.println(dwarf.getName());
        System.out.println(dwarf.getHealthPoints());
        dwarf.setHealthPoints(99);
        System.out.println(dwarf.getHealthPoints());
        
        Weapon weapon = new Weapon("Test", 100, 100);
        System.out.println(weapon.getName());
        System.out.println(weapon.getAttackSpeed());
        System.out.println(weapon.getSize());
        weapon.setSize(99);
        System.out.println(weapon.getSize());
        
        Animal animal = new Animal("Test", 100, 4);
        System.out.println(animal.getName());
        System.out.println(animal.getSize());
        System.out.println(animal.getAge());
        animal.setAge(99);
        System.out.println(animal.getAge());
	}
}
```

# 1.3. побочный эффект передачи по ссылке
```java
public class Main
{
	public static void main(String[] args) {
        Dwarf dwarf = new Dwarf("Test", 100);
        System.out.println(dwarf.getName());
        System.out.println(dwarf.getHealthPoints());
        dwarf.setHealthPoints(99);
        System.out.println(dwarf.getHealthPoints());
        
        Dwarf another_dwarf = dwarf;
        another_dwarf.setHealthPoints(0);

        // Теперь у певого дфаврфа тоже 0 здоровья
        System.out.println(dwarf.getHealthPoints());
	}
}
```
