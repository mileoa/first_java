class Dwarf {
    protected String name;
    protected int healthPoints;

    public Dwarf(String name, int healthPoints) {
        this.name = name;
        this.healthPoints = healthPoints;
    }

    public String getName() {
        return this.name;
    }

    public int getHealthPoints() {
        return this.healthPoints;
    }

    public void setHealthPoints(int newHealthPoints) {
        if (newHealthPoints >= 0) {
            this.healthPoints = newHealthPoints;
        }
    }
    
    public void sing() {
        System.out.println("La-La-La");
    }
    
    public void sleep() {
        this.healthPoints = this.healthPoints + 1;
    }

}

class Weapon {
    protected int size;
    protected int attackSpeed;
    protected String name;

    public Weapon(String name, int size, int attackSpeed) {
        this.name = name;
        this.size = size;
        this.attackSpeed = attackSpeed;
    }

    public int getSize() {
        return this.size;
    }

    public String getName() {
        return this.name;
    }

    public int getAttackSpeed() {
        return this.attackSpeed;
    }

    public void setSize(int size) {
        this.size = size;
    }
    
    public void sharpen() {
        this.size = this.size - 1;
        this.attackSpeed = this.attackSpeed + 1;
    }

}

class Animal {
    protected int size;
    protected String name;
    protected int age;

    public Animal(String name, int size, int age) {
        this.name = name;
        this.size = size;
        this.age = age;
    }

    public int getSize() {
        return this.size;
    }

    public String getName() {
        return this.name;
    }

    public int getAge() {
        return this.age;
    }

    public void setAge(int age) {
        this.age = age;
    }
    
    public void feed() {
        this.size = this.size + 1;
    }

}

class Dog extends Animal {

    public Dog(String name, int size, int age) {
        super(name, size, age);    
        }

    public void feed() {
        this.size = this.size + 2;
    }

    public String sound() {
        return "Woof";
    }    

}

class Cat extends Animal {

    public Cat(String name, int size, int age) {
        super(name, size, age);    
    }

    public String sound() {
        return "Murr";
    }    

}
